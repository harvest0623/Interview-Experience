## 四、手撕代码
1. 写一个防抖函数

``` JavaScript
function debounce(fn, delay) {
    let timer = null;
    return function (...args) {
        if (timer) {
            clearTimeout(timer);
        }
        timer = setTimeout(() => {
            fn.apply(this, args);
        }, delay);
    }
}

const Name = debounce((name) => {
    console.log(name);
}, 1000);
Name('张三');
Name('李四');
Name('王五');
```
